2.4GHz Jammer

Creating a 2.4GHz signal jammer with ESP32 and NRF24 chip (Educational Purposes Only)

He has some antennas and asked me if had any nrf24 modules and told him that had some e01 ml01dp5’s laying around. So we worked together to make a signal jammer. Since had 3 of the transceiver modules, my friend built a 2 antenna jammer and built a single antenna jammer.

Creating Circuit

We started off making the PCBs. Our power source would be 4 aa batteries, making a total of 6v. Since the esp32 we were using cannot take 6v and the transceiver modules can only take up to 3.6v, we used an Lm1117t to regulate the voltage to 3.3v.

For the circuit, the battery is connected to a switch, and when the switch is on, the current flows to a 10uf and .1uf capacitor in parallel. The current continues flowing to the 3.3v regulator and comes out of the regulator with the less voltage. This voltage is sent both to the transceiver modules, the esp32, a 100 uf capacitor and a .1uf capacitor.

Midst of building the circuit! Here, you can see how I've attatched the 7 LEDs.

Controlling Channels

Since there are up to 125 channels on the 2.4 GHz bandwidth, there must be a way to control the channel the jammer is jamming. Since an 8 segment display with enough digits didn't fit on the PCB, we had to use another method of telling numbers. My friend came up with using binary with LEDs. We calculated that we need 7 LEDs to cover all 125 channels.

We also needed a way to control which channel and to do that, we decided to use a potentiometer.

Problems

During soldering, ran into a few problems. First was making sure that the GPIO pins I used worked for the purpose. I had to plan accordingly to make sure the LEDs were using pins that were not read only and that the potentiometer should be on a read only pin.

After finishing soldering, I still had a few problems. My esp32 wouldn’t turn on. Turns out there was a few places where forgot to solder, upon fixing all the circuit issues, I was ready to upload the code using arduino ide.

The Code

A lot of the code that actually codes for signal jamming was found on a github. I had to modify it slightly so I can set channels and display the channel to the LEDs.


  #include <SPI.h>
  #include <RF24.h>
  #include <ezButton.h>
  #include "esp_bt.h"
  #include "esp_wifi.h"
  #include <WiFi.h>
  #define CE_PIN  22
  #define CSN_PIN 21
  SPIClass *sp = nullptr;
  RF24 radio(CE_PIN, CSN_PIN, 16000000);  /// CAN SET SPI SPEED TO 16000000 BY DEFAULT ITS 10000000
  byte i = 45;  ///CHANNEL NRF NEEDS TO START  37-50 CHANNEL

  unsigned int flag = 0;


  ezButton toggleSwitch(33);



  const uint8_t SAMPLES_PER_CHANNEL = 20;
  const uint8_t DETECTION_THRESHOLD = 6; // must trigger 6+ times
  // List of LED pins (7 LEDs for 7 bits)
  int ledPins[] = {13, 17, 27, 26, 25, 33, 32};
  int numLeds = sizeof(ledPins) / sizeof(ledPins[0]);

  // Potentiometer pin
  const int potPin = 35; // GPIO 35

  void setup() {
    Serial.begin(115200); // for debugging


    // Initialize LED pins as OUTPUT
    for (int i = 0; i < numLeds; i++) {
      pinMode(ledPins[i], OUTPUT);
      digitalWrite(ledPins[i], LOW); // start all LEDs OFF
    }
    WiFi.mode(WIFI_STA);
    WiFi.disconnect(true);
    delay(100);

    Serial.println("Wi-Fi scan starting...");
  int networks = WiFi.scanNetworks(false, true);

    if (networks == 0) {
      Serial.println("No networks found");
    } else {
      Serial.printf("%d networks found\n", networks);
      for (int i = 0; i < networks; i++) {
        Serial.printf(
          "%2d | %-25s | RSSI: %4d dBm | CH: %2d | %s\n",
          i + 1,
          WiFi.SSID(i).c_str(),
          WiFi.RSSI(i),
          WiFi.channel(i),
          (WiFi.encryptionType(i) == WIFI_AUTH_OPEN) ? "OPEN" : "SECURED"
        );
      }
    }

    WiFi.scanDelete();


  /* _---------------------------------------------------------------------------*/
    esp_bt_controller_deinit();
    esp_wifi_stop();
    esp_wifi_deinit();
    Serial.begin(115200);
    toggleSwitch.setDebounceTime(50);
    initSP();
  }

  void loop(void) {
    // Read pot and scale 0-4095 → 0-127
    int potValue = map(analogRead(potPin), 0, 4095, 0, 125);

    displayBinary(potValue); // update LEDs
   // Serial.println(potValue);

    toggleSwitch.loop();  // MUST call the loop() function first
    /* if (toggleSwitch.isPressed())

      Serial.println("two");*/

    int state = toggleSwitch.getState();
    if(potValue>14){
      one();
    }else{
      radio.setChannel(potValue);
    }


  }

  // Function to display an integer 0-127 on 7 LEDs
  void displayBinary(int num) {
    for (int i = 0; i < numLeds; i++) {
      int bit = (num >> i) & 0x01; // extract bit i
      digitalWrite(ledPins[i], bit);
    }
  }



  void initSP() {
    sp = new SPIClass(VSPI);
    sp->begin();
    if (radio.begin(sp)) {
      delay(200);
      Serial.println("Sp Started !!!");
      radio.setAutoAck(false);
      radio.stopListening();
      radio.setRetries(0, 0);
      radio.setPayloadSize(5);   ////SET VALUE ON RF24.CPP
      radio.setAddressWidth(3);  ////SET VALUE ON RF24.CPP
      radio.setPALevel(RF24_PA_MAX, true);
      radio.setDataRate(RF24_2MBPS);
      radio.setCRCLength(RF24_CRC_DISABLED);
      radio.printPrettyDetails();
      radio.startConstCarrier(RF24_PA_MAX, i);  ////EDITED VALUES ON LIBRARY ....SET 5 BYTES PAYLOAD SIZE//REDUCE RF24_MAX TO RF24_HIGH OR LOW IF RF NOT STABLE



    } else {
      Serial.println("SP couldn't start !!!");
    }
  }
  void two() {

    ///CHANNEL WITH 2 SPACING HOPPING
    if (flag == 0) {
      i += 2;
    } else {
      i -= 2;
    }

    if ((i > 79) && (flag == 0)) {
      flag = 1;
    } else if ((i < 2) && (flag == 1)) {
      flag = 0;
    }

    radio.setChannel(i);
    //Serial.println(i);
  }

  void one() {
    ////SWEEP CHANNEL
    for (int i = 0; i < 79; i++) {
      radio.setChannel(i);
    }

  }
